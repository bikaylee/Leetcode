public String removeDuplicates(String s) {
    Deque<Character> stack = new ArrayDeque<>();
    for (char c : s.toCharArray()) {
        if (stack.isEmpty() || stack.peekLast() != c)
            stack.addLast(c);
        else
            stack.removeLast();
    }

    StringBuilder ans = new StringBuilder();
    while (!stack.isEmpty())
        ans.append(stack.removeFirst());

    return ans.toString();
}
